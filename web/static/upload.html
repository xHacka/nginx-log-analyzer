Example JSON line format

{
          "time"       : "$msec",                 // epoch seconds with milliseconds
          "remote_addr": "$remote_addr",          // client IP address
          "host"       : "$host",                 // requested host/domain
          "method"     : "$request_method",       // HTTP method
          "path"       : "$uri",                  // URL path without query
          "q"          : "$args",                 // query string arguments
          "protocol"   : "$server_protocol",      // HTTP protocol version
          "status"     : "$status",               // response status code
          "bytes"      : "$body_bytes_sent",      // response size in bytes
          "city"       : "$geo_city_name",        // geoip city (if available)
          "country"    : "$geo_country_code",     // geoip country code
          "user_agent" : "$http_user_agent"       // client user agent
}
